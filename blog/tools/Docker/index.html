    # Docker

    # docker常用命令

    # 查看帮助
docker --help
# 搜索镜像
docker search [镜像名称]
# 获取镜像
docker pull [镜像名称]:[tag]
# 推送镜像到仓库
docker push [镜像名称]:[tag]
# 登录镜像仓库地址
docker [login/logout] [仓库地址]
# 查看本地镜像
docker images
# 删除本地镜像
docker rmi [镜像名称/镜像ID]
# 删除本地镜像 - 强制
docker rmi -f [镜像名称/镜像ID]
# 查看镜像详情
docker inspect [镜像名称/镜像ID]
